public static Map<String,Set<Id>> mappingFunction(){
       Map<String,Set<Id>> mapping = new Map<String,Set<Id>>();
       for(Class__c con1 : [SELECT Id, Name,(Select Id From Students__r) FROM Class__c]){
           Set<id> iss = new Set<id>();
           for(Student__c con : [ SELECT Id FROM Student__c WHERE Class__r.Name=: con1.Name]){
               iss.add(con.id);
           }
           mapping.put(con1.Name, iss);
       }
       return mapping;
   }
